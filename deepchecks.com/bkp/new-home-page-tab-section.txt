                        function initProgressSlider($section) {
                            const $slider = $section.find('.progress-tab-slider');
                            const $items = $section.find('.item');
                            let percentTime = 0;
                            let tick;
                            let progressIndex = 0;
                            const intervalSpeed = 20;
                            const progressSpeedFactor = 1 / 5.1;

                            // Prevent double init
                            if (!$slider.hasClass('slick-initialized')) {
                                $slider.slick({
                                    infinite: true,
                                    dots: false,
                                    autoplay: false,
                                    speed: 1000,
                                    arrows: false,
                                    slidesToShow: 1,
                                    slidesToScroll: 1,
                                    accessibility: false
                                });
                            }

                            // Set up progress bars
                            $section.find('.progressBarj').each(function (index) {
                                if (!$(this).find('.inProgressj').length) {
                                    $(this).html(`<div class="inProgressj inProgressj${index}"></div>`);
                                }
                            });

                            function resetProgress() {
                                clearInterval(tick);
                                $section.find('.inProgressj').css({ height: '0%' });
                                $items.removeClass('active-progress');
                                $items.find('.card-info').slideUp();
                            }

                            function startProgress() {
                                resetProgress();
                                percentTime = 0;
                                tick = setInterval(runProgress, intervalSpeed);
                            }

                            function runProgress() {
                                const $currentSlide = $slider.find(`.slick-slide[data-slick-index="${progressIndex}"]`);
                                if ($currentSlide.attr('aria-hidden') === "true") {
                                    progressIndex = $slider.find('.slick-slide[aria-hidden="false"]').data('slickIndex');
                                    startProgress();
                                    return;
                                }

                                percentTime += progressSpeedFactor;
                                $section.find(`.inProgressj${progressIndex}`).css({ height: `${percentTime}%` });
                                $section.find(`.item-progress${progressIndex}`).addClass('active-progress')
                                    .find('.card-info').slideDown();

                                if (percentTime >= 100) {
                                    $slider.slick('slickNext');
                                    const totalSlides = $slider.find('.slick-slide').not('.slick-cloned').length;
                                    progressIndex = (progressIndex + 1) % totalSlides;
                                    startProgress();
                                }
                            }

                            // Click to navigate manually
                            $items.off('click').on('click', function () {
                                if (!$(this).hasClass('active-progress')) {
                                    clearInterval(tick);
                                    progressIndex = $(this).find('[data-slick-index]').data('slickIndex');
                                    $slider.slick('slickGoTo', progressIndex);
                                    startProgress();
                                }
                            });

                            // Start progress initially
                            startProgress();

                            // Store interval ID so it can be cleared on tab change
                            $section.data('progressInterval', tick);
                        }

                        function destroyProgressTimers() {
                            $('.progress-tab-slider-section').each(function () {
                                const tick = $(this).data('progressInterval');
                                if (tick) {
                                    clearInterval(tick);
                                }
                            });
                        }

                        $(document).ready(function () {
                            // Initialize first tab
                            initProgressSlider($('.tab-pane.active .progress-tab-slider-section'));

                            // On Bootstrap tab switch
                            $('button[data-toggle="tab"]').on('shown.bs.tab', function () {
                                destroyProgressTimers(); // stop any existing timers
                                const targetPane = $($(this).data('target'));
                                setTimeout(() => {
                                    initProgressSlider(targetPane.find('.progress-tab-slider-section'));
                                }, 10); // slight delay to ensure DOM rendered
                            });
                        });
                    </script>
